---
title: "Get Reward"
api: "GET /rewards/{id}"
description:
  "This endpoint returns the reward corresponding to the ID in the path."
---

### Parameters

<ParamField path="id" type="string" required>
  The reward ID
</ParamField>

### Response

<ResponseField name="id" type="string">
  Reward ID
</ResponseField>

<ResponseField name="user_id" type="string">
  User ID
</ResponseField>

<ResponseField name="merchant_id" type="string">
  Merchant ID
</ResponseField>

<ResponseField name="campaign_id" type="string">
  Campaign ID
</ResponseField>

<ResponseField name="state" type="enum">
  Reward State (`available`, `active`, `used`)
</ResponseField>

<ResponseField name="fine_print" type="string">
  Fine print for the reward
</ResponseField>

<ResponseField name="instructions" type="string">
  Staff instructions for the reward
</ResponseField>

<ResponseField name="available_at" type="string">
  Time the reward was given to the user in ISO8601-format
</ResponseField>

<ResponseField name="activated_at" type="string">
  Time the reward was activated in ISO8601-format
</ResponseField>

<ResponseField name="used_at" type="string">
  Time the reward was marked as used in ISO8601-format
</ResponseField>

<ResponseField name="retire_at" type="string">
  Time the reward will be automatically retired in ISO8601-format
</ResponseField>

<ResponseField name="coupon_code" type="string">
  This hash will be populated if the reward is 'active' or 'used' and the
  merchant uses coupon codes.
</ResponseField>

<ResponseField name="coupon_code.code" type="hash">
  The raw coupon code value that should be rendered based on the
  `coupon_code_type` enum
</ResponseField>

<ResponseField name="coupon_code.type" type="enum">
  If set, should be displayed instead of the raw coupon code value
</ResponseField>

<ResponseField name="coupon_code.display" type="string">
  If set, should be displayed instead of the raw coupon code value
</ResponseField>

<ResponseField name="earn" type="string">
  Details of how the reward was earned
</ResponseField>

<ResponseField name="earn.text" type="string">
  Description of how the reward was earned
</ResponseField>

<ResponseField name="redeem" type="string">
  Details of how the reward can be redeemed
</ResponseField>

<ResponseField name="redeem.type" type="string">
  How the reward can be redeemed (`manual`, `automatic`)
</ResponseField>

<ResponseField name="redeem.text" type="string">
  Description of what the reward is
</ResponseField>

<ResponseField name="redeem.venue" type="string">
  Where the reward can be used: (`instore`, `online`, `all`)
</ResponseField>

<ResponseField name="redeem.window" type="string">
  Number of minutes the reward is available for use after activation
</ResponseField>
