---
title: "Get Communication Settings"
api: "GET /communication_settings"
description:
  "This endpoint returns a user's communication settings. The notification key
  reflects a user's settings for receiving push notifications in their app or
  via text if they don't have an app installed. The email key reflects a user's
  settings for receiving emails."
---

### Parameters

<ParamField body="merchant_id" type="string">
  Merchant ID
</ParamField>

### Response

<ResponseField name="id" type="string">
  The ID of the settings record
</ResponseField>

<ResponseField name="merchant_id" type="string">
  The ID of the merchant
</ResponseField>

<ResponseField name="user_id" type="string">
  The ID of the user
</ResponseField>

<ResponseField name="reward_earned" type="hash">
  Settings for when a user earns a loyalty reward
</ResponseField>

<ResponseField name="reward_earned.notification" type="boolean">
  app notification setting
</ResponseField>

<ResponseField name="reward_earned.email" type="boolean">
  email setting
</ResponseField>

<ResponseField name="reward_unused" type="hash">
  Settings for when a user has an unused reward
</ResponseField>

<ResponseField name="reward_unused.notification" type="boolean">
  app notification setting
</ResponseField>

<ResponseField name="reward_unused.email" type="boolean">
  email setting
</ResponseField>

<ResponseField name="reward_progress" type="hash">
  Settings for when a user earns loyalty progress
</ResponseField>

<ResponseField name="reward_progress.notification" type="boolean">
  app notification setting
</ResponseField>

<ResponseField name="reward_progress.email" type="boolean">
  email setting
</ResponseField>

<ResponseField name="reward_offer" type="hash">
  Settings for when a merchant sends an offer
</ResponseField>

<ResponseField name="reward_offer.notification" type="boolean">
  app notification setting
</ResponseField>

<ResponseField name="reward_offer.email" type="boolean">
  email setting
</ResponseField>

<ResponseField name="feedback_available" type="hash">
  Settings for when a user has the opportunity to leave feedback for a purchase
</ResponseField>

<ResponseField name="feedback_available.notification" type="boolean">
  app notification setting
</ResponseField>

<ResponseField name="marketing_general" type="hash">
  Settings for when a merchant sends general marketing
</ResponseField>

<ResponseField name="marketing_general.email" type="boolean">
  email setting
</ResponseField>

<RequestExample>

```bash
curl https://api.thanxsandbox.com/communication_settings/woerihfslkwer \
  -X PATCH \
  $AUTH_HEADERS
  -d '{
    "communication_settings": {
      "reward_earned": {
        "notification": false,
        "email": false
      },
      "feedback_available": {
        "notification": true
      }
    }
  }'
```

</RequestExample>
